```javascript
// Landing Page Data
const landingPageData = {
  heroSection: {
    title: "Welcome to our website",
    subtitle: "This is a sample landing page",
    backgroundImage: "https://example.com/background-image.jpg",
    callToAction: "Learn More"
  },
  featuresSection: {
    title: "Features",
    features: [
      {
        icon: "https://example.com/icon1.png",
        title: "Feature 1",
        description: "This is feature 1"
      },
      {
        icon: "https://example.com/icon2.png",
        title: "Feature 2",
        description: "This is feature 2"
      },
      {
        icon: "https://example.com/icon3.png",
        title: "Feature 3",
        description: "This is feature 3"
      }
    ]
  },
  testimonialsSection: {
    title: "Testimonials",
    testimonials: [
      {
        image: "https://example.com/testimonial1.jpg",
        name: "John Doe",
        testimonial: "This is a great website"
      },
      {
        image: "https://example.com/testimonial2.jpg",
        name: "Jane Doe",
        testimonial: "I love this website"
      },
      {
        image: "https://example.com/testimonial3.jpg",
        name: "Bob Smith",
        testimonial: "This website is amazing"
      }
    ]
  },
  callToActionSection: {
    title: "Get Started",
    callToAction: "Sign Up Now"
  }
};

// Function to render the landing page
function renderLandingPage(data) {
  const heroSection = document.getElementById("hero-section");
  const featuresSection = document.getElementById("features-section");
  const testimonialsSection = document.getElementById("testimonials-section");
  const callToActionSection = document.getElementById("call-to-action-section");

  // Render hero section
  heroSection.innerHTML = `
    <h1>${data.heroSection.title}</h1>
    <p>${data.heroSection.subtitle}</p>
    <button>${data.heroSection.callToAction}</button>
    <img src="${data.heroSection.backgroundImage}" alt="Background Image">
  `;

  // Render features section
  featuresSection.innerHTML = `
    <h2>${data.featuresSection.title}</h2>
    <div class="features">
      ${data.featuresSection.features.map(feature => `
        <div class="feature">
          <img src="${feature.icon}" alt="Feature Icon">
          <h3>${feature.title}</h3>
          <p>${feature.description}</p>
        </div>
      `).join("")}
    </div>
  `;

  // Render testimonials section
  testimonialsSection.innerHTML = `
    <h2>${data.testimonialsSection.title}</h2>
    <div class="testimonials">
      ${data.testimonialsSection.testimonials.map(testimonial => `
        <div class="testimonial">
          <img src="${testimonial.image}" alt="Testimonial Image">
          <h3>${testimonial.name}</h3>
          <p>${testimonial.testimonial}</p>
        </div>
      `).join("")}
    </div>
  `;

  // Render call to action section
  callToActionSection.innerHTML = `
    <h2>${data.callToActionSection.title}</h2>
    <button>${data.callToActionSection.callToAction}</button>
  `;
}

// Render the landing page
renderLandingPage(landingPageData);
```
